栈和队列

Stack Overview 堆栈概述

  1. The first ADT we saw today was the stack. A stack is a LIFO (last-in, first-out) data structure. It generally supports a limited set of operations: 我们今天看到的第一个抽象数据类型是栈。栈是一种后进先出（ LIFO ）数据结构。它通常只支持有限的操作：
Member Function 成员函数Description 描述
clear()removes all elements from the stack 从栈中移除所有元素
equals(stack)returns true if the two stacks contain the same elements in the same order, false otherwise 如果两个栈包含相同顺序的相同元素，则返回 true，否则返回 false
isEmpty()returns true if the stack contains no elements, false otherwise 如果栈不包含任何元素，则返回 true，否则返回 false
peek()returns the element on the top of the stack without removing it 返回栈顶元素但不移除它
pop()removes the element from the top of the stack and returns it 从栈顶移除元素并返回它
push(value)pushes value onto the stack 将值压入栈中
size()returns the number of values in the stack 返回栈中的元素数量
. . .For an exhaustive list, see: Stanford Stack class 如需完整列表，请参阅：斯坦福栈类

Queue Overview 队列概述

  1. We then discussed the queue ADT. A queue is a FIFO (first-in, first-out) data structure. Like the stack, it generally supports a limited set of operations: 我们接着讨论了队列抽象数据类型。队列是一种先进先出（FIFO）的数据结构。和栈一样，它通常支持一组有限的操作：
Member Function 成员函数Description 描述
clear()removes all elements from the queue 从队列中移除所有元素
dequeue()removes and returns the frontmost element in the queue 移除并返回队列中最前面的元素
enqueue(value)adds value to the back of the queue 将 value 添加到队列的末尾
equals()returns true if the two queues contain the same elements in the same order, false otherwise 如果两个队列包含相同顺序的相同元素，则返回 true，否则返回 false
isEmpty()returns true if the queue contains no elements, false otherwise 如果队列不包含任何元素，则返回 true，否则返回 false
peek()returns the frontmost element in the queue without removing it 返回队列中最前面的元素，但不移除它
size()returns the number of values in the queue 返回队列中的值数量

#include <iostream>
#include "console.h"
#include "SimpleTest.h"
#include "strlib.h"
using namespace std;

string extractAlpha(string s)
{
    string result;

    for (int i = 0; i < s.length(); i++)
    {
        if (isalpha(s[i]))
        {
            result += s[i];
        }
    }
    return result;
}

bool passwordChecksOut(string username, string password)
{
    string alphaPortion = extractAlpha(password);
    return !stringContains(toLowerCase(alphaPortion), toLowerCase(username));
}

STUDENT_TEST("various tests of extractAlpha() function")
{
    EXPECT_EQUAL(extractAlpha("sean"), "sean");
    EXPECT_EQUAL(extractAlpha("sean11"), "sean");
    EXPECT_EQUAL(extractAlpha("1sean"), "sean");
    EXPECT_EQUAL(extractAlpha("s1ean"), "sean");
    EXPECT_EQUAL(extractAlpha("s1e1a1n"), "sean");
    EXPECT_EQUAL(extractAlpha("s9e$a***&n"), "sean");
    EXPECT_EQUAL(extractAlpha(""), "");
    EXPECT_EQUAL(extractAlpha("9$&"), "");
}

STUDENT_TEST("这是我的测试, 关于函数 passwordChecksOut()")
{
    EXPECT_EQUAL(passwordChecksOut("dk","d9327309k**%dd"), false);
    EXPECT_EQUAL(passwordChecksOut("dk","kdbusa*7^//"),true);
}

int main_test()
{

    string username = "Sean";
    string password = "s1e1a1n1'1s1p1a1s1s1w1o1r1d";

        cout << "Hooray!" << endl;
    }

    {
        cout << "Oh no! Bad password." << endl;
    }

    return 0;
}

void printVector(Vector<int>& v)

#include <iostream>
#include "console.h"
#include "SimpleTest.h"
#include "string.h"
#include "vector.h"
using namespace std;

int widthOfNum(int num){
    if (num == 0) return 1; // 0是1位数

    int count = 0;
    // 处理负数：转换为正数
    if (num < 0) {
        num = -num;
        ++count;
    }

    while (num > 0) {
        count++;
        num /= 10;
    }
    return count;
}

int maxWidthOfVector(Vector<int> &v){
    int numVec = v.size();
    int maxNum = widthOfNum(v[0]);
    for(int i= 1;i<numVec;i++){
        if(widthOfNum(v[i]) > maxNum)
        {maxNum = widthOfNum(v[i]);}
    }
    return maxNum;
}



void printVectorAdvanced(Vector<int> &v)
{
    //runSimpleTests(SELECTED_TESTS);
    int maxVecWidth = maxWidthOfVector(v)+2;
    string s1,s2,s3;
    int count = 0;
    for(int i:v)
    {
        int iWidth = widthOfNum(i);
        s1 += string("+") + string(maxVecWidth,'-');
        int whiteSpace1 = (maxVecWidth-iWidth)/2;
        int whiteSpace2 = maxVecWidth - whiteSpace1-iWidth;
        s2 += string("|") + string(whiteSpace1,' ') + to_string(i) + string(whiteSpace2,' ');
        int whiteSpace3 = (maxVecWidth+1)/2;
        s3 += string(whiteSpace3,' ') + to_string(count) + string(maxVecWidth/2,' ');
        count++;
    }
    s1 += string("+");
    s2 += string("|");
    cout << s1 << "\n" << s2 << "\n" << s1 << "\n" << s3 << endl;
}

STUDENT_TEST("测试maxWidthOfVector函数")
{
    Vector<int> v1 = {43,1,9,129,5821,3};
    Vector<int> v2 = {43,1,9,129,582,3,-256};
    EXPECT_EQUAL(maxWidthOfVector(v1), 4);
    EXPECT_EQUAL(maxWidthOfVector(v2), 7);
}


STUDENT_TEST("测试函数widthOfNum"){
    EXPECT_EQUAL(widthOfNum(1890),4);
    EXPECT_EQUAL(widthOfNum(0),1);
    EXPECT_EQUAL(widthOfNum(-1),2);
}

int main()
{
    //runSimpleTests(SELECTED_TESTS);
    Vector<int> v1 = {43,1,9,129,5821,3};
    Vector<int> v2 = {43,1,9,129,5821,3,-5633};
